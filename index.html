function procesarNotas(id) { const input = document.getElementById(id).value.trim(); if (!input) return []; // 1. Separamos por ESPACIOS para identificar notas diferentes // 2. En cada nota, cambiamos la coma por punto para que sea decimal válido return input.split(/\s+/) .map(n => parseFloat(n.replace(',', '.'))) .filter(n => !isNaN(n)); } function calcular() { const notasCuaderno = procesarNotas('cuaderno'); const notasParciales = procesarNotas('parciales'); const notasTrimestral = procesarNotas('trimestral'); // REGLA: 10% (0.10) por CADA nota de cuaderno const puntosCuaderno = notasCuaderno.reduce((acc, n) => acc + (n * 0.10), 0); // REGLA: 15% (0.15) por CADA nota de parcial const puntosParciales = notasParciales.reduce((acc, n) => acc + (n * 0.15), 0); // REGLA: 50% (0.50) de la nota del trimestral (solo la primera) const notaTrimestral = notasTrimestral.length > 0 ? notasTrimestral[0] : 0; const puntosTrimestral = notaTrimestral * 0.50; // Suma total const notaFinal = puntosCuaderno + puntosParciales + puntosTrimestral; // Mostrar resultado const divRes = document.getElementById('resultado'); divRes.style.display = 'block'; divRes.innerHTML = `Nota Final: ${notaFinal.toFixed(2)}`; // Colores: Verde si es >= 5, rojo si es < 5 divRes.className = notaFinal >= 5 ? 'aprobado' : 'suspenso'; }
